```java
public class AllOOP {
    int v;
    static int count;

    AllOOP() { this(0); }
    AllOOP(int v) { this.v = v; count++; }

    static void show() { System.out.println(count); }
    void show(int x) { System.out.println(v + x); }

        new AllOOP().show(5);
        new AllOOP(7).show();
        AllOOP.show();
```

3. STRING · STRINGBUFFER · STRINGTOKENIZER · DATE
-------------------------------------------------
```java
import java.util.*;

public class AllStringDate {
    public static void main(String[] args) {
        String s = "10/12/2025";
        StringTokenizer st = new StringTokenizer(s, "/");
        StringBuffer sb = new StringBuffer();
        while (st.hasMoreTokens()) sb.append(st.nextToken());
        System.out.println(sb);

        Date d = new Date();
        System.out.println(d.getDate() + "/" + (d.getMonth() + 1) + "/" + (d.getYear() + 1900));
    }
}
```

```java
abstract class A {
    final int MAX = 100;
    void show() { System.out.println("A"); }
    abstract void demo();
}
class B extends A {
    void show() { super.show(); System.out.println("B"); }
    void demo() {}
    public static void main(String[] a) {
        A ref = new B();
        ref.show();
        System.out.println(ref.MAX);
    }
}
```

7. MULTITHREADING – LIFE-CYCLE · PRIORITY · SYNCHRONISATION · INTER-THREAD
-------------------------------------------------
```java
class Q {
    int n; boolean valueSet = false;
    synchronized void put(int i) {
        while (valueSet) try { wait(); } catch (Exception e) {}
        n = i; valueSet = true; notify();
    }
    synchronized int get() {
        while (!valueSet) try { wait(); } catch (Exception e) {}
        valueSet = false; notify(); return n;
    }
}
class Producer extends Thread {
    Q q;
    Producer(Q q) { this.q = q; setPriority(MAX_PRIORITY); }
    public void run() { for (int i = 0; i < 5; i++) q.put(i); }
}
class Consumer extends Thread {
    Q q;
    Consumer(Q q) { this.q = q; setPriority(MIN_PRIORITY); }
    public void run() { for (int i = 0; i < 5; i++) System.out.println(q.get()); }
}
public class AllThread {
    public static void main(String[] a) {
        Q q = new Q();
        new Producer(q).start();
        new Consumer(q).start();
    }
}
```
